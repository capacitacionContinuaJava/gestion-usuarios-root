CREATE TABLE rhtuser (
	user_id numeric NOT NULL,
	user_name varchar NOT NULL,
	status bool NOT NULL,
	create_user_id varchar NOT NULL,
	create_date timestamp NOT NULL,
	modified_user_id varchar NULL,
	modified_date timestamp NULL,
	person_id numeric NOT NULL,
	CONSTRAINT rhtuser_pk PRIMARY KEY (user_id),
	CONSTRAINT rhtuser_fk FOREIGN KEY (person_id) REFERENCES public.rhtpersona(person_id)
);
